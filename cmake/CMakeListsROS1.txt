add_compile_options(-Wall -Wextra -pedantic -Werror)
add_definitions(-DUSING_ROS_1)

find_package(catkin REQUIRED COMPONENTS
  event_array_msgs
  roscpp
  rosbag
  )
  
catkin_package()


include_directories(
  include
  ${catkin_INCLUDE_DIRS}
)

#
# --------- encoding/decoding library
#
add_library(evt_library src/evt_3_utils.cpp)
target_link_libraries(evt_library ${catkin_LIBRARIES})

#
# --------- performance tool
#
add_executable(perf src/perf_ros1.cpp)
target_link_libraries(perf ${catkin_LIBRARIES})

# --------- conversion tools

add_executable(bag_to_raw src/bag_to_raw_ros1.cpp)
target_link_libraries(bag_to_raw evt_library ${catkin_LIBRARIES})

add_executable(raw_to_bag src/raw_to_bag_ros1.cpp src/evt_3_utils.cpp)
target_link_libraries(raw_to_bag ${catkin_LIBRARIES})

#############
## Install ##
#############

install(TARGETS perf bag_to_raw raw_to_bag
  RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION})
